## Endpoint

```
GET /api/usuarios
```

## Authentication

This endpoint requires authentication. Include a valid JWT token in the Authorization header.

```
Authorization: Bearer YOUR_JWT_TOKEN
```

<Warning>
  This is an admin endpoint. Only authenticated users with valid permissions should access this endpoint.
</Warning>

## Response

Returns an array of user profile objects, ordered by creation date (most recent first).

<ResponseField name="users" type="array">
  Array of user profile objects
  
  <Expandable title="User Profile Object">
    <ResponseField name="id" type="string">
      Unique identifier for the user (UUID from Supabase Auth)
    </ResponseField>
    
    <ResponseField name="nombre_completo" type="string">
      Full name of the user
    </ResponseField>
    
    <ResponseField name="email" type="string">
      User's email address
    </ResponseField>
    
    <ResponseField name="rol" type="string">
      User role - `ESTUDIANTE` (Student), `USER` (Default), or admin roles
    </ResponseField>
    
    <ResponseField name="carrera" type="string">
      User's academic program or career (optional)
    </ResponseField>
    
    <ResponseField name="genero" type="string">
      User's gender (optional)
    </ResponseField>
    
    <ResponseField name="documento" type="string">
      User's document/ID number (optional)
    </ResponseField>
    
    <ResponseField name="codigo_institucional" type="string">
      Institutional code for students (optional, only for ESTUDIANTE role)
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp of when the user profile was created
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Timestamp of the last profile update
    </ResponseField>
  </Expandable>
</ResponseField>

## Notes

<Note>
  This endpoint returns all users ordered by `created_at` in descending order (most recent first).
</Note>

<Info>
  The endpoint uses the `admin_supabase` client on the backend to bypass Row Level Security policies and retrieve all user profiles.
</Info>

## Implementation Details

The endpoint is implemented in the usuario controller:

```typescript
export const obtenerUsuarios = async (req: any, res: any) => {
  try {
    const { data, error } = await admin_supabase
      .from('perfil_usuario')
      .select('*')
      .order('created_at', { ascending: false })

    if (error) throw error

    res.json(data)
  } catch (err: any) {
    res.status(400).json({ error: err.message })
  }
}
```
