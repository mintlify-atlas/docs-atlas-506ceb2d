---

## Overview

This endpoint retrieves a list of all conferences registered in the CONIITI 2026 system. Each conference includes detailed information about the room (sala), event day (dia_evento), and associated speakers (ponentes) through the ponencia_ponente relationship.

<Note>
  This is a public endpoint and does not require authentication.
</Note>

## Request

No parameters required.

## Response

<ResponseField name="data" type="array">
  Array of conference objects
  
  <Expandable title="Conference Object">
    <ResponseField name="id" type="string">
      Unique identifier for the conference
    </ResponseField>
    
    <ResponseField name="titulo" type="string">
      Title of the conference
    </ResponseField>
    
    <ResponseField name="descripcion" type="string">
      Description of the conference content
    </ResponseField>
    
    <ResponseField name="fecha_inicio" type="datetime">
      Start date and time of the conference
    </ResponseField>
    
    <ResponseField name="fecha_fin" type="datetime">
      End date and time of the conference
    </ResponseField>
    
    <ResponseField name="sala_id" type="string">
      Reference to the room where the conference will take place
    </ResponseField>
    
    <ResponseField name="dia_evento_id" type="string">
      Reference to the event day
    </ResponseField>
    
    <ResponseField name="sala" type="object">
      Nested object containing room information
      
      <Expandable title="Room Object">
        <ResponseField name="id" type="string">
          Room identifier
        </ResponseField>
        
        <ResponseField name="nombre" type="string">
          Room name
        </ResponseField>
        
        <ResponseField name="capacidad" type="integer">
          Room capacity
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="dia_evento" type="object">
      Nested object containing event day information
      
      <Expandable title="Event Day Object">
        <ResponseField name="id" type="string">
          Event day identifier
        </ResponseField>
        
        <ResponseField name="fecha" type="date">
          Date of the event day
        </ResponseField>
        
        <ResponseField name="descripcion" type="string">
          Description of the event day
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="ponencia_ponente" type="array">
      Array of speaker relationships
      
      <Expandable title="Speaker Relationship">
        <ResponseField name="ponente" type="object">
          Speaker information
          
          <Expandable title="Speaker Object">
            <ResponseField name="id" type="string">
              Speaker identifier
            </ResponseField>
            
            <ResponseField name="nombre" type="string">
              Speaker name
            </ResponseField>
            
            <ResponseField name="biografia" type="string">
              Speaker biography
            </ResponseField>
            
            <ResponseField name="email" type="string">
              Speaker email address
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Example Request

<CodeGroup>

```bash cURL
curl -X GET https://api.coniiti2026.com/api/ponencias \
  -H "Content-Type: application/json"
```

```javascript JavaScript
const response = await fetch('https://api.coniiti2026.com/api/ponencias', {
  method: 'GET',
  headers: {
    'Content-Type': 'application/json'
  }
});

const data = await response.json();
console.log(data);
```

```python Python
import requests

response = requests.get('https://api.coniiti2026.com/api/ponencias')
data = response.json()
print(data)
```
